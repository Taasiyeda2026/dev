/** * דשבורד כספים 2026 - קובץ שרת מלא (Code.gs) */ const SHEET_NAMES = { settings: 'הגדרות', expenses: 'הוצאות', incomes: 'גיול', normalizedExpenses: 'הוצאותמסודר', flow: 'תזרים', summary: 'סיכום', lists: 'רשימות' }; const EXPENSE_MONTH_COLUMNS = { 3: 3, 4: 4, 5: 5, 6: 6, 7: 7, 8: 8 }; const MONTH_LABELS = { 1: 'ינואר', 2: 'פברואר', 3: 'מרץ', 4: 'אפריל', 5: 'מאי', 6: 'יוני', 7: 'יולי', 8: 'אוגוסט', 9: 'ספטמבר', 10: 'אוקטובר', 11: 'נובמבר', 12: 'דצמבר' }; // --- פונקציות תשתית --- function include(filename) { return HtmlService.createHtmlOutputFromFile(filename).getContent(); } function doGet() { return HtmlService .createTemplateFromFile('index') .evaluate() .setTitle('דשבורד כספים - תעשיידע') .addMetaTag('viewport', 'width=device-width, initial-scale=1') .setXFrameOptionsMode(HtmlService.XFrameOptionsMode.ALLOWALL); } // --- פונקציית הליבה שמושכת את כל הנתונים --- function getDashboardData() { try { const settings = readSettings_(); const months = buildMonthRange_(settings.startMonth, settings.endMonth); // קריאת נתונים גולמיים מהגיליונות const expensesSheetRows = readExpenseEntries_(months); const incomesSheetRows = readIncomeEntries_(settings.timezone); // עיבוד הנתונים לתצוגה const monthlySummary = buildMonthlySummaryFromRaw_(months, expensesSheetRows, incomesSheetRows, settings.openingBalance); const cashflowDays = buildCashflowDaysFromRaw_( months, expensesSheetRows, incomesSheetRows, settings.openingBalance, settings.workYear, settings.balanceResetDate, settings.manualOpeningBalance ); const weekly = buildWeeklyFromCashflow_(cashflowDays); const totalIncome = monthlySummary.reduce((sum, r) => sum + (Number(r.income) || 0), 0); const totalExpense = monthlySummary.reduce((sum, r) => sum + (Number(r.expense) || 0), 0); const closingBalance = monthlySummary.length ? monthlySummary[monthlySummary.length - 1].closing : settings.openingBalance; return { organizationName: settings.organizationName, workYear: settings.workYear, openingBalance: settings.openingBalance, months: months, totalIncome: totalIncome, totalExpense: totalExpense, closingBalance: closingBalance, monthlySummary: monthlySummary, cashflowDays: cashflowDays, weekly: weekly, expensesSheetRows: expensesSheetRows, incomesSheetRows: incomesSheetRows, startMonth: settings.startMonth, endMonth: settings.endMonth, balanceResetDate: settings.balanceResetDate ? formatDateLabel_(settings.balanceResetDate) : '', manualOpeningBalance: settings.manualOpeningBalance, metadata: { cashflowSource: 'derived-from-income-expense-sheets', generatedAt: new Date().toISOString() } }; } catch (e) { throw new Error("שגיאה בעיבוד הנתונים: " + e.message); } } // --- פונקציות קריאה מהגיליונות (Helper Functions) --- function getRequiredSheet_(sheetName) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(sheetName); if (!sheet) throw new Error("לא נמצא גיליון בשם: " + sheetName); return sheet; } function readSettings_() { const sheet = getRequiredSheet_(SHEET_NAMES.settings); const data = sheet.getRange(1, 1, sheet.getLastRow(), 2).getValues(); const map = {}; data.forEach(r => { if(r[0]) map[r[0]] = r[1]; }); return { workYear: map['שנת עבודה'] || 2026, openingBalance: Number(map['יתרת פתיחה']) || 0, balanceResetDate: parseSettingsDate_(map['תאריך איפוס יתרה']), manualOpeningBalance: parseOptionalNumber_(map['יתרת פתיחה ידנית']), startMonth: Number(map['חודש התחלה']) || 3, endMonth: Number(map['חודש סיום']) || 8, organizationName: map['שם ארגון'] || 'תעשיידע', timezone: Session.getScriptTimeZone() }; } function readExpenseEntries_(months) { const sheet = getRequiredSheet_(SHEET_NAMES.expenses); const values = sheet.getDataRange().getValues(); const results = []; // מדלגים על כותרת (שורה 1) for (let i = 1; i < values.length; i++) { const row = values[i]; const day = row[0]; const category = row[1]; const detail = row[8]; const group = row[9] || 'כללי'; months.forEach(m => { const colIdx = EXPENSE_MONTH_COLUMNS[m] - 1; const amount = Number(row[colIdx]); if (amount && amount !== 0) { results.push({ id: `exp-${i+1}-${m}`, row: i + 1, month: m, day: day, category: category, detail: detail, group: group, amount: amount }); } }); } return results; } function readIncomeEntries_(timezone) { const sheet = getRequiredSheet_(SHEET_NAMES.incomes); const values = sheet.getDataRange().getValues(); const results = []; for (let i = 1; i < values.length; i++) { const row = values[i]; if (!row[2]) continue; // אם אין סכום, דלג results.push({ id: `inc-${i+1}`, row: i + 1, account: row[0], detail: row[1], amount: Number(row[2]), dueDateLabel: Utilities.formatDate(new Date(row[3]), timezone, "dd/MM/yyyy"), dueDateISO: row[3] instanceof Date ? row[3].toISOString().split('T')[0] : "", notes: row[4], fund: row[5], month: row[3] instanceof Date ? row[3].getMonth() + 1 : null }); } return results; } function buildMonthlySummaryFromRaw_(months, expenses, incomes, openingBalance) { let currentBalance = openingBalance; return months.map(m => { const monthIncome = incomes.filter(inc => inc.month === m).reduce((s, r) => s + r.amount, 0); const monthExpense = expenses.filter(exp => exp.month === m).reduce((s, r) => s + r.amount, 0); const closing = currentBalance + monthIncome - monthExpense; const row = { month: m, label: MONTH_LABELS[m], income: monthIncome, expense: monthExpense, closing: closing, balance: closing }; currentBalance = closing; return row; }); } function buildCashflowDaysFromRaw_(months, expenses, incomes, openingBalance, workYear, balanceResetDate, manualOpeningBalance) { const allowedMonths = new Set((months || []).map(Number)); const year = Number(workYear) || new Date().getFullYear(); const byDay = new Map(); (incomes || []).forEach(inc => { const month = toMonthNumber_(inc.month); if (!month || !allowedMonths.has(month)) return; const dateObj = parseDateFromIncome_(inc, year, month); if (!dateObj) return; const key = toIsoDateKey_(dateObj); if (!byDay.has(key)) { byDay.set(key, createEmptyFlowDay_(key, month, dateObj)); } const day = byDay.get(key); const amount = toNumber_(inc.amount); if (!amount) return; day.incomeAmount += amount; if (inc.detail) day.incomeDescParts.push(normalizeText_(inc.detail)); }); (expenses || []).forEach(exp => { const month = toMonthNumber_(exp.month); if (!month || !allowedMonths.has(month)) return; const dayNumber = Number(exp.day); if (!Number.isFinite(dayNumber) || dayNumber < 1 || dayNumber > 31) return; const dateObj = new Date(year, month - 1, dayNumber); if (dateObj.getMonth() + 1 !== month) return; const key = toIsoDateKey_(dateObj); if (!byDay.has(key)) { byDay.set(key, createEmptyFlowDay_(key, month, dateObj)); } const day = byDay.get(key); const amount = toNumber_(exp.amount); if (!amount) return; day.expenseAmount += amount; const detailText = normalizeText_(exp.detail || exp.category || ''); if (detailText) day.expenseDescParts.push(detailText); }); const sorted = [...byDay.values()].sort((a, b) => a.sortKey.localeCompare(b.sortKey)); let runningBalance = toNumber_(openingBalance); const resetKey = balanceResetDate ? toIsoDateKey_(balanceResetDate) : ''; const hasManualReset = Boolean(resetKey) && Number.isFinite(Number(manualOpeningBalance)); const manualStartBalance = Number(manualOpeningBalance); let manualResetApplied = false; return sorted.map((row, index) => { if (hasManualReset && !manualResetApplied && row.sortKey >= resetKey) { runningBalance = manualStartBalance; manualResetApplied = true; } runningBalance += row.incomeAmount - row.expenseAmount; return { id: `flow-${index + 1}`, date: formatDateLabel_(row.dateObj), month: row.month, incomeDesc: dedupeAndJoin_(row.incomeDescParts), incomeAmount: row.incomeAmount, expenseDesc: dedupeAndJoin_(row.expenseDescParts), expenseAmount: row.expenseAmount, balance: runningBalance }; }); } function buildWeeklyFromCashflow_(cashflowDays) { const byWeek = new Map(); (cashflowDays || []).forEach(day => { const month = toMonthNumber_(day.month); const dateParts = parseDateLabel_(day.date); if (!month || !dateParts) return; const week = Math.floor((dateParts.day - 1) / 7) + 1; const key = `${month}-w${week}`; if (!byWeek.has(key)) { byWeek.set(key, { key: key, month: month, week: week, startDay: dateParts.day, endDay: dateParts.day, startDate: day.date, endDate: day.date, income: 0, expense: 0, details: [] }); } const weekItem = byWeek.get(key); weekItem.startDay = Math.min(weekItem.startDay, dateParts.day); weekItem.endDay = Math.max(weekItem.endDay, dateParts.day); if (dateParts.day <= weekItem.startDay) weekItem.startDate = day.date; if (dateParts.day >= weekItem.endDay) weekItem.endDate = day.date; weekItem.income += toNumber_(day.incomeAmount); weekItem.expense += toNumber_(day.expenseAmount); if (day.incomeDesc) { weekItem.details.push({ date: day.date, type: 'income', detail: day.incomeDesc, amount: toNumber_(day.incomeAmount) }); } if (day.expenseDesc) { weekItem.details.push({ date: day.date, type: 'expense', detail: day.expenseDesc, amount: toNumber_(day.expenseAmount) }); } }); return [...byWeek.values()] .sort((a, b) => (a.month - b.month) || (a.week - b.week)) .map(item => ({ key: item.key, month: item.month, week: item.week, label: `שבוע ${item.week}`, range: `${item.startDate} - ${item.endDate}`, income: item.income, expense: item.expense, details: item.details })); } function buildMonthRange_(start, end) { const range = []; for (let i = start; i <= end; i++) range.push(i); return range; } // --- פונקציות עדכון (Update/Delete) --- function updateRowData(type, data) { const sheetName = type === 'expense' ? SHEET_NAMES.expenses : SHEET_NAMES.incomes; const sheet = getRequiredSheet_(sheetName); const rowIdx = parseInt(data.id.split('-')[1]); if (type === 'expense') { sheet.getRange(rowIdx, 1).setValue(data.day); sheet.getRange(rowIdx, 2).setValue(data.category); sheet.getRange(rowIdx, 9).setValue(data.detail); const col = EXPENSE_MONTH_COLUMNS[data.month]; sheet.getRange(rowIdx, col).setValue(data.amount); } else { sheet.getRange(rowIdx, 1).setValue(data.account); sheet.getRange(rowIdx, 2).setValue(data.detail); sheet.getRange(rowIdx, 3).setValue(data.amount); sheet.getRange(rowIdx, 4).setValue(new Date(data.dueDate)); sheet.getRange(rowIdx, 5).setValue(data.notes); sheet.getRange(rowIdx, 6).setValue(data.fund); } return { ok: true }; } function deleteExpenseRow(id) { const rowIdx = parseInt(id.split('-')[1]); getRequiredSheet_(SHEET_NAMES.expenses).deleteRow(rowIdx); return { ok: true }; } function deleteIncomeRow(id) { const rowIdx = parseInt(id.split('-')[1]); getRequiredSheet_(SHEET_NAMES.incomes).deleteRow(rowIdx); return { ok: true }; } function createEmptyFlowDay_(key, month, dateObj) { return { sortKey: key, month: month, dateObj: dateObj, incomeAmount: 0, expenseAmount: 0, incomeDescParts: [], expenseDescParts: [] }; } function toNumber_(value) { const num = Number(value); return Number.isFinite(num) ? num : 0; } function toMonthNumber_(value) { const month = Number(value); return Number.isFinite(month) && month >= 1 && month <= 12 ? month : null; } function parseDateFromIncome_(incomeRow, fallbackYear, fallbackMonth) { if (incomeRow && incomeRow.dueDateISO) { const date = new Date(incomeRow.dueDateISO); if (!isNaN(date.getTime())) return date; } if (incomeRow && incomeRow.dueDateLabel) { const parts = String(incomeRow.dueDateLabel).split('/'); if (parts.length === 3) { const day = Number(parts[0]); const month = Number(parts[1]); const year = Number(parts[2]); const date = new Date(year, month - 1, day); if (!isNaN(date.getTime())) return date; } } const date = new Date(fallbackYear, fallbackMonth - 1, 1); return isNaN(date.getTime()) ? null : date; } function toIsoDateKey_(date) { const y = date.getFullYear(); const m = String(date.getMonth() + 1).padStart(2, '0'); const d = String(date.getDate()).padStart(2, '0'); return `${y}-${m}-${d}`; } function formatDateLabel_(date) { const d = String(date.getDate()).padStart(2, '0'); const m = String(date.getMonth() + 1).padStart(2, '0'); const y = date.getFullYear(); return `${d}/${m}/${y}`; } function normalizeText_(value) { return String(value || '') .replace(/[\r\n]+/g, ' ') .replace(/\s{2,}/g, ' ') .trim(); } function dedupeAndJoin_(list) { const cleaned = (list || []).map(normalizeText_).filter(Boolean); const unique = [...new Set(cleaned)]; return unique.join(', '); } function parseDateLabel_(label) { const match = String(label || '').match(/^(\d{2})\/(\d{2})\/(\d{4})$/); if (!match) return null; return { day: Number(match[1]), month: Number(match[2]), year: Number(match[3]) }; } function parseSettingsDate_(value) { if (!value) return null; if (value instanceof Date && !isNaN(value.getTime())) { return new Date(value.getFullYear(), value.getMonth(), value.getDate()); } const normalized = String(value).trim(); const slash = normalized.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/); if (slash) { const date = new Date(Number(slash[3]), Number(slash[2]) - 1, Number(slash[1])); return isNaN(date.getTime()) ? null : date; } const date = new Date(normalized); if (isNaN(date.getTime())) return null; return new Date(date.getFullYear(), date.getMonth(), date.getDate()); } function parseOptionalNumber_(value) { if (value === '' || value === null || value === undefined) return NaN; const num = Number(value); return Number.isFinite(num) ? num : NaN; }
